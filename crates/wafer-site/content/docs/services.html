Clients

Blocks access services (database, storage, crypto, etc.) by calling other blocks via ctx.call_block(). The wafer_run::clients module provides typed wrapper functions that make this easy. WASM blocks use the equivalent typed clients from the Guest SDK.

Database

CRUD operations on collections with filtering, pagination, and raw SQL:

use wafer_run::clients::database as db;

// Create
let record = db::create(ctx, "users", json!({"name": "Alice", "role": "admin"}))?;

// Read
let user = db::get(ctx, "users", "user-id-123")?;

// List with filtering and pagination
let opts = db::ListOptions {
    filters: vec![db::Filter {
        field: "role".into(),
        operator: db::FilterOp::Equal,
        value: json!("admin"),
    }],
    limit: 20,
    ..Default::default()
};
let users = db::list(ctx, "users", &opts)?;

// Aggregation
let count = db::count(ctx, "users", &opts.filters)?;

// Raw SQL (for complex queries)
let results = db::query_raw(ctx, "SELECT * FROM users WHERE role = ?", &["admin"])?;

Storage

Binary object storage organized into folders:

use wafer_run::clients::storage as store;

// Create a folder
store::create_folder(ctx, "avatars")?;

// Upload
store::put(ctx, "avatars", "alice.png", &bytes, "image/png")?;

// Download
let obj = store::get(ctx, "avatars", "alice.png")?;

// List objects in a folder
let files = store::list(ctx, "avatars", &opts)?;

// Manage folders
let folders = store::list_folders(ctx)?;

Crypto

Password hashing and JWT token management:

use wafer_run::clients::crypto;

// Password hashing (argon2)
let hash = crypto::hash(ctx, "my-password")?;
crypto::compare_hash(ctx, "my-password", &hash)?;

// JWT tokens
let token = crypto::sign(ctx, &claims, 3600)?;  // 1 hour expiry
let decoded = crypto::verify(ctx, &token)?;

// Random bytes
let bytes = crypto::random_bytes(ctx, 32)?;

Network

Make outbound HTTP requests from blocks:

use wafer_run::clients::network;

let resp = network::do_request(
    ctx,
    "GET",
    "https://api.example.com/data",
    &headers,
    None,
)?;

Logger

Structured logging with optional key-value fields:

use wafer_run::clients::logger;

logger::info(ctx, "Request processed");
logger::error_with(ctx, "Failed to connect", &[("service", "database")]);

Config

Read and write configuration key-value pairs:

use wafer_run::clients::config;

let val = config::get(ctx, "api_key")?;
let val_or_default = config::get_default(ctx, "api_key", "fallback");
config::set(ctx, "last_sync", "2025-01-01T00:00:00Z")?;

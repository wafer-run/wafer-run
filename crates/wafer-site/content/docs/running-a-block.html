Running a Block

Once you have a block, you need a wafer runtime to host it. The runtime (wafer-run) is a pure block/flow executor with no transport knowledge. For HTTP, the wafer-core crate provides an @wafer/http-listener block (enable the http feature) that converts HTTP requests into messages and dispatches them to flows. The Go and Node.js runtimes load pre-built WASM blocks and expose a run() API you wire into your own HTTP framework.

Install

npm install wafer-run
go get github.com/wafer-run/wafer-run-go
# Cargo.toml
[dependencies]
wafer-run = "0.1"
wafer-core = { version = "0.1", features = ["http"] }
serde_json = "1"
tokio = { version = "1", features = ["full"] }

Minimal Server

A complete runnable example — registers a block, defines a flow, and starts an HTTP server on port 8080:

const { WaferRuntime } = require("wafer-run");
const http = require("http");

const w = new WaferRuntime();
w.register("hello", "./hello.wasm");
w.register("main", "./main-flow.json");
w.resolve();
w.start();

// Wire into Node's HTTP server
http.createServer((req, res) => {
    const chunks = [];
    req.on("data", (c) => chunks.push(c));
    req.on("end", () => {
        const body = Buffer.concat(chunks).toString();
        const msg = JSON.stringify({
            kind: req.method + ":" + req.url,
            data: body,
            meta: { "http.method": req.method, "http.path": req.url }
        });
        const result = JSON.parse(w.run("main", msg));
        const status = result.action === "error" ? 500 : 200;
        res.writeHead(status, { "Content-Type": "application/json" });
        res.end(result.response?.data ?? "{}");
    });
}).listen(8080);

console.log("listening on :8080");
package main

import (
    "fmt"
    "io"
    "net/http"

    wafer "github.com/wafer-run/wafer-run-go"
)

func main() {
    w := wafer.New()
    defer w.Close()

    w.Register("hello", "./hello.wasm")
    w.Register("main", "./main-flow.json")
    w.Resolve()
    w.Start()

    // Wire into net/http
    http.HandleFunc("/", func(rw http.ResponseWriter, r *http.Request) {
        body, _ := io.ReadAll(r.Body)
        msg := wafer.NewMessage(r.Method+":"+r.URL.Path, body)
        msg.SetMeta("http.method", r.Method)
        msg.SetMeta("http.path", r.URL.Path)

        result := w.Run("main", msg)
        rw.Header().Set("Content-Type", "application/json")
        if result.IsError() {
            rw.WriteHeader(500)
        }
        if result.Response != nil {
            rw.Write(result.Response.Data)
        }
    })

    fmt.Println("listening on :8080")
    http.ListenAndServe(":8080", nil)
}
use wafer_run::*;

#[tokio::main]
async fn main() {
    let mut w = Wafer::new();
    wafer_core::register_all(&mut w);

    // Rust can define blocks inline — no .wasm needed
    w.register_block_func("hello", |_ctx, msg| {
        json_respond(msg, &serde_json::json!({"hello": "world"}))
    });

    // Configure the HTTP listener block
    w.add_block_config("@wafer/http-listener", serde_json::json!({
        "flow": "main",
        "listen": "0.0.0.0:8080",
    }));

    let flow: FlowDef = serde_json::from_str(r#"{
        "id": "main",
        "root": { "block": "hello" }
    }"#).unwrap();
    w.add_flow_def(&flow);

    let w = w.start().unwrap();
    println!("listening on :8080");
    tokio::signal::ctrl_c().await.unwrap();
    w.shutdown();
}

Run It

Start the server and test with curl:

node index.js
go run .
cargo run

Then in another terminal:

curl http://localhost:8080/
# {"hello":"world"}

Running a Block

Once you have a block, you need a wafer runtime to host it. The Rust runtime supports native blocks and includes a built-in HTTP bridge. The Go and Node.js runtimes load pre-built WASM blocks and expose an execute() API you wire into your own HTTP framework.

Install

npm install wafer-run
go get github.com/wafer-run/wafer-run-go
# Cargo.toml
[dependencies]
wafer-run = { version = "0.1", features = ["full"] }
serde_json = "1"
tokio = { version = "1", features = ["full"] }
axum = "0.8"

Minimal Server

A complete runnable example — registers a block, defines a chain, and starts an HTTP server on port 8080:

const { WaferRuntime } = require("wafer-run");
const http = require("http");

const w = new WaferRuntime();
w.registerWasmBlock("hello", "./hello.wasm");
w.addChainDef(JSON.stringify({
    id: "main",
    root: { block: "hello" },
    http: { routes: [{ path: "/", path_prefix: true }] }
}));
w.resolve();
w.start();

// Wire into Node's HTTP server
http.createServer((req, res) => {
    const chunks = [];
    req.on("data", (c) => chunks.push(c));
    req.on("end", () => {
        const body = Buffer.concat(chunks).toString();
        const msg = JSON.stringify({
            kind: req.method + ":" + req.url,
            data: body,
            meta: { "http.method": req.method, "http.path": req.url }
        });
        const result = JSON.parse(w.execute("main", msg));
        const status = result.action === "error" ? 500 : 200;
        res.writeHead(status, { "Content-Type": "application/json" });
        res.end(result.response?.data ?? "{}");
    });
}).listen(8080);

console.log("listening on :8080");
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"

    wafer "github.com/wafer-run/wafer-run-go"
)

func main() {
    w := wafer.New()
    defer w.Close()

    w.RegisterWASMBlock("hello", "./hello.wasm")
    w.AddChain(wafer.ChainDef{
        ID:   "main",
        Root: wafer.NodeDef{Block: "hello"},
        HTTP: &wafer.HTTPRouteDef{
            Routes: []wafer.HTTPRoute{{Path: "/", PathPrefix: true}},
        },
    })
    w.Resolve()
    w.Start()

    // Wire into net/http
    http.HandleFunc("/", func(rw http.ResponseWriter, r *http.Request) {
        body, _ := io.ReadAll(r.Body)
        msg := wafer.NewMessage(r.Method+":"+r.URL.Path, body)
        msg.SetMeta("http.method", r.Method)
        msg.SetMeta("http.path", r.URL.Path)

        result := w.Execute("main", msg)
        rw.Header().Set("Content-Type", "application/json")
        if result.IsError() {
            rw.WriteHeader(500)
        }
        if result.Response != nil {
            rw.Write(result.Response.Data)
        }
    })

    fmt.Println("listening on :8080")
    http.ListenAndServe(":8080", nil)
}
use wafer_run::*;
use std::sync::Arc;

#[tokio::main]
async fn main() {
    let mut w = Wafer::new();

    // Rust can define blocks inline — no .wasm needed
    w.register_block_func("hello", |_ctx, msg| {
        json_respond(msg.clone(), 200, &serde_json::json!({"hello": "world"}))
    });

    let chain: ChainDef = serde_json::from_str(r#"{
        "id": "main",
        "root": { "block": "hello" },
        "http": { "routes": [{ "path": "/", "path_prefix": true }] }
    }"#).unwrap();
    w.add_chain_def(&chain);
    w.resolve().unwrap();
    w.start().unwrap();

    // Built-in HTTP bridge auto-registers chain routes
    let app = wafer_run::bridge::auto_register(Arc::new(w));
    let listener = tokio::net::TcpListener::bind("0.0.0.0:8080").await.unwrap();
    println!("listening on :8080");
    axum::serve(listener, app).await.unwrap();
}

Run It

Start the server and test with curl:

node index.js
go run .
cargo run

Then in another terminal:

curl http://localhost:8080/
# {"hello":"world"}
